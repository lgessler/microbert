كومپيۇتېر يۇمشاق ماتېرىيالىنى قوغداش نىزامى جۇڭخۇا خەلق جۇمھۇرىيىتى گوۋۇيۈەنىنىڭ پەرمانى

339 -نومۇرلۇق كومپيۇتېر يۇمشاق ماتېرىيالىنى قوغداش نىزامى ئېلان قىلىندى، 2002-يىل 1-ئاينىڭ 1-كۈنىدىن باشلاپ يولغا قويۇلىدۇ. زۇڭلى جۇرۇڭجى 2001-يىل 12-ئاينىڭ 20-كۈنى

بىرىنچى باب ئومۇمىي پرىنسىپ 1 -ماددا بۇ نىزام كومپيۇتېر يۇمشاق ماتېرىيالى ئەسەر ھوقۇقى ئىگىسىنىڭ ھوقۇق-مەنپەئىتىنى قوغداش، كومپيۇتېر يۇمشاق ماتېرىيالىنى ئېچىش، تارقىتىش ۋە ئىشلىتىشتە كېلىپ چىققان مەنپەئەت مۇناسىۋىتىنى تەڭشەش، كومپيۇتېر يۇمشاق ماتېرىيالىنى ئېچىشقا ۋە ئىشلىشىتشكە ئىلھام بېرىپ، يۇمشاق ماتېرىيال سانائىتى ۋە ئىشلىتىشكە ئىلھام بېرىپ، يۇمشاق ماتېرىيال سانائىتى ۋە خەلق ئىگىلىكىنى ئۇچۇرلاشتۇرۇش تەرەققىياتىنى ئىلگىرى سۈرۈش مەقسىتىدە، «جۇڭخۇا خەلق جومھورىيىتىنىڭ "ئەسەر ھوقۇقى قانۇنى" غا ئاساسەن تۈزۈپ چىقىلدى.

2 -ماددا بۇ نىزامدا ئېيتىلغان كومپيۇتېر يۇمشاق ماتېرىيالى )تۆۋەندە يۇمشاق ماتېرىيال دېيلىدۇ( كومپيۇتېر پروگراممىسى ۋە شۇنىڭغا ئالاقىدار ئارخىپنى كۆرسىتىدۇ. 3 -ماددا بۇ نىزامدىكى تۆۋەندىكى سۆز-ئىبارىلەرنىڭ مەنىسى:

(1)كومويۇتېر پروگراممىسى مەلۇم نەتىجىگە ئېرىشىش ئۈچۈن كومپيۇتېر قاتارلىق ئۇچۇر بىر تەرەپ قىلىش ئىقتىدارىغا ئىگە قۇرۇلما ئارقىلىق ئىجرا قىلىنىدىغان شىفىرلاشقان بۇيرۇق تەرتىپى، ئاپتوماتىك ھالدا شىفىرلاشقان بۇيرۇق تەرتىپىگە ئايلاندۇرغىلى بولىدىغان بەلگىلەشتۈرۈلگەن بۇيرۇق تەرتىپى ۋە ياكى بەلگىلەشتۈرۈلگەن جۈملە تەرتىپىنى كۆرسىتىدۇ. بىر كومپيۇتېر پروگراممىسىنىڭ ئەسلىي پروگراممىسى بىلەن نىشان پروگراممىسى بىر ئەسەر بولىدۇ؛ (2) ئارخىپ پروگرامما - مەزمۇنى، تەركىبى، لايىھىلىنىشى، ئىقتىدار ئۆلچىمى، ئېچىش ئەھۋالى، سىناش نەتىجىسى ۋە ئىشلىتىش ئۇسۇلىنى تەسۋىرلەيدىغان يازما ماتېرىيال، چېرتىيو قاتارلىقلارنى كۆرسىتىدۇ، مەسىلەن، پروگرامما لايىھىسى چۈشەندۈرۈشى، ئىش تەرتىپى سخېمىسى، ئىشلەتكۈچىلەر قوللانمىسى ۋە باشقىلار؛ (3) يۇمشاق ماتېرىيال ئاچقۇچى - ئېچىشقا ئەمەلىي ئۇيۇشتۇرغان، بىۋاسىتە ئاچقان ھەمدە ئېچىپ بولۇنغان يۇمشاق ماتېرىيال مەسئۇلىيىتىنى ئۈستىگە ئالغان قانۇنىي ئىگە ياكى باشقا تەشكىلات ۋە ياكى ئۆزىدىكى شارائىتقا تايىنىپ يۇمشاق ماتېرىيالنى مۇستەقىل ئاچقان ھەمدە يۇمشاق ماتېرىيال مەسئۇلىيىتىنى ئۈستىگە ئالغان تەبىئىي شەخسنى كۆرسىتىدۇ؛ (4)يۇمشاق ماتېرىيال ئەسەر ھوقۇقى ئىگىسى - مۇشۇ نىزامدىكى بەلگىلىمە بويىچە يۇمشاق ماتېرىيالنىڭ ئەسەر ھوقۇقىدىن بەھرىمەن بولىدىغان تەبىئىي شەخس، قانۇنىي ئىگە ۋە باشقا تەشكىلاتلارنى كۆرسىتىدۇ.

4 -ماددا بۇ نىزام بويىچە قوغدىلىدىغان يۇمشاق ماتېرىيال ئاچقۇچى مۇستەقىل ئاچقان ھەمدە مەلۇم شەكىللىك جىسىمغا مۇقىملاشتۇرغان يۇمشاق ماتېرىيال بولۇشى شەرت.

5-ماددا جۇڭگو پۇقراسى، قانۇنىي ئىگە ياكى باشقا تەشكىلات ئېلان قىلىنغان ياكى قىلىنمىغان بولۇشىدىن قەتئىينەزەر، مۇشۇ نىزام بويىچە ئۆزى ئاچقان يۇمشاق ماتېرىيالنىڭ ئەسەر ھوقۇقىدىن بەھرىمەن بولىدۇ. يۇمشاق ماتېرىيالى ئالدى بىلەن جۇڭگو چېگرىسى ئىچىدە تارقىتىلغان چەت ئەللىكلەر، دۆلەت تەۋەلىكى يوقلار مۇشۇ نىزام بويىچە ئەسەر ھوقۇقىدىن بەھرىمەن بولىدۇ.

چەت ئەللىكلەر، دۆلەت تەۋەلىكى يوقلارنىڭ يۇمشاق ماتېرىيالى ئۇنى ئاچقۇچى تەۋە دۆلەت ياكى دائىم تۇرۇشلۇق دۆلەت جۇڭگو بىلەن ئىمزالىغان كېلىشىم بويىچە ياكى جۇڭگو قاتناشقان خەلقئارالىق شەرتنامە بويىچە بەھرىمەن بولىدىغان ەئسەر ھوقۇقى بۇ نىزام تەرىپىدىن قوغدىلىدۇ.

6-ماددا بۇ نىزامدا قوغدىلىدىغان ئەسەر ھوقۇقى يۇمشاق ماتېرىيالىنى ئېچىش ئىدىيىسىنى بىر تەرەپ قىلىش جەريانى، مەشغۇلات ئۇسۇلى ياكى ماتېماتىكىلىق ئۇقۇم قاتارلىقلارغا كېڭەيتىلمەيدۇ.

7 -ماددا يۇمشاق ماتېرىيال ئەسەر ھوقۇقى ئىگىسى گوۋۇيۈەننىڭ ئەسەر ھوقۇقىنى مەمۇرىي باشقۇرۇش تارمىقى ئېتىراپ قىلغان يۇمشاق ماتېرىيال رويخەتكە ئېلىش ئورگىنىغا بېرىپ، رويخەتكە ئالدۇرسا بولىدۇ. يۇمشاق ماتېرىيال رويخەتكە ئېلىش ئورگىنى بەرگەن رويخەتكە ئېلىش ھۆججىتى رويخەتكە ئېلىش ئىشىنىڭ دەسلەپكى ئىسپاتى بولىدۇ.

يۇمشاق ماتېرىيالنى روىەتكە ئالدۇرۇشتا پۇل تاپشۇرۇش كېرەك. يۇمشاق ماتېرىيالنى رويخەتكە ئېلىش ھەق ئۆلچىمىنى گوۋۇيۈەننىڭ ئەسەر ھوقۇقىنى مەمۇرىي باشقۇرۇش تارمىقى گوۋۇيۈەننىڭ باھا باشقۇرۇش تارمىقى بىلەن بىرلىكتە بەلگىلەيدۇ. ئىككىنچى باب يۇمشاق ماتېرىيال ئەسەر ھوقۇقى 8 -ماددا يۇمشاق ماتېرىيال ئەسەر ھوقۇقى ئىگىسى تۆۋەندىكى ھوقۇقلاردىن بەھرىمەن بولىدۇ: (1)ئېلان قىلىش ھوقۇقى، يەنى يۇمشاق ماتېرىيالنى جامائەتكە ئېلان قىلىش ياكى قىلماسلىقنى بەلگىلەش ھوقۇقى؛ (2)ئىسمىنى يېزىش ھوقۇقى، يەنى ئاچقۇچىنىڭ سالاھىيىتىنى بىلدۈرۈپ يۇمشاق ماتېرىيالىغا ئىسمىنى يېزىش ھوقۇقى؛ (3)تۈزىتىش ھوقۇقى، يەنى يۇمشاق ماتېرىيال مەزمۇنىنى تولۇقلاش، مەزمۇن چىقىرىۋېتىش ياكى بۇيرۇق، جۈملە تەرتىپىنى ئۆزگەرتىش ھوقۇقى؛ (4)كۆپەيتىش ھوقۇقى، يەنى يۇمشاق ماتېرىيالنى بىر نۇسخا ياكى كۆپ نۇسخا قىلىپ ئىشلەش ھوقۇقى؛ (5) تارقىتىش ھوقۇقى، يەنى سېتىش ياكى تەقدىم قىلىش ئۇسۇلى ئارقىلىق جامائەتنى يۇمشاق ماتېرىيالنىڭ ئەسلىي نۇسخىسى ياكى كۆپەيتمىسى بىلەن تەمىنلەش ھوقۇقى؛ (6) ئىجارىگە بېرىش ھوقۇقى، يەنى يۇمشاق ماتېرىيالنى باشقىلارنىڭ ۋاقىتلىق ئىشلىتىشىگە ھەقلىق ئىجازەت بېرىش ھوقۇقى، لېكىن يۇمشاق ماتېرىيالنى ئىجارىگە بېرىش ئاساسىي ئۆلچەم قىلىنمىغانلىرى بۇنىڭ سىرتىدا؛ (7) ئۇچۇر تورىدا تارقىتىش ھوقۇقى، يەنى جامائەتنى سىملىق ياكى سىمسىز شەكىلدە يىۇمشاق ماتېرىيال بىلەن تەمىنلەپ، تاللىغان ۋاقىتتا ۋە ئورۇندا يۇمشاق ماتېرىيالغا ئىگە قىلىش ھوقۇقى؛ (8) تەرجىمە قىلىش ھوقۇقى، يەنى ئەسلىدىكى يۇمشاق ماتېرىيالنى بىرخىل تەبىئىي تىل-يېزىقتىن يەنە بىر خىل تەبىئىي تىل-يېشىققا ئۆرۈش ھوقۇقى؛ (9) يۇمشاق ماتېرىيال ئەسەر ھوقۇقى ئىگىسى بەھرىمەن بولۇشقا تېگىشلىك باشقا ھوقۇقلار.

يۇمشاق ماتېرىيال ئەسەر ھوقۇقى ئىگىسى ئۆزىنىڭ يۇمشاق ماتېرىيال ئەسەر ھوقۇقىنى باشقىلارنىڭ يۈرگۈزۈشىگە ئىجازەت بەرسە بولىدۇ ھەمدە ھەق ئېلىشقا ھوقۇقلۇق.

يۇمشاق ماتېرىيال ئەسەر ھوقۇقى ئىگىسى ئۆزىنىڭ يۇمشاق ماتېرىيال ئەسەر ھوقۇقىنىڭ ھەممىسى ياكى بىر قىسمىنى ئۆتۈنۈپ بەرسە بولىدۇ ھەمدە ھەق ئېلىشقا ھوقۇقلۇق.

9 -ماددا يۇمشاق ماتېرىيال ئەسەر ھوقۇقى يۇمشاق ماتېرىيالنى ئاچقۇچىغا مەنسۇپ، بۇ نىزامدا باشقا بەلگىلىمە بولغانلىرى بۇنىڭ سىرتىدا. قارشى ئىسپات تېپىلمىسا، يۇمشاق ماتېرىيالغا ئىسمى يېزىلغان تەبىئىي شەخس، قانۇنىي ئىگە ياكى باشقا تەشكىلات ئاچقۇچى بولىدۇ.

10 -ماددا ئىككىدىن ئارتۇق تەبىئىي شەخس، قانۇنىي ئىگە ياكى باشقا تەشكىلاتلار بىرلىشىپ ئاچقان يۇمشاق ماتېرىيال ئەسەر ھوقۇقىنىڭ تەۋەلىكىنى بىرلىشىپ ئاچقۇچىلار يازما توختام ئىمزالاپ پۈتۈشىدۇ. يازما توختام بولمىسا ياكى توختامدا ئېنىق پۈتۈشمىگەن بولسا، بىرلىشىپ ئاچقان يۇمشاق ماتېرىيالنىڭ ئايرىپ ئىشلەتكىلى بولىدىغانلىرىدا، ئاچقۇچىلار ئۆزى ئاچقان قىسمىنىڭ ئەسەر ھوقۇقىدىن مۇستەقىل بەھرىمەن بولسا بولىدۇ. لېكىن ئەسەر ھوقۇقىنى يۈرگۈزگەندە، ئۇ ھوقۇقنى بىرلىشىپ ئاچقان يۇمشاق ماتېرىيالنىڭ بىر پۈتۈن ئەسەر ھوقۇقىغا كېڭەيتىشكە بولمايدۇ. بىرلىشىپ ئاچقان يۇمشاق ماتېرىيالنى ئايرىپ ئىشلەتكىلى بولمىسا، ئۇنىڭ ئەسەر ھوقۇقىدىن بىرلىشىپ ئاچقۇچىلار ئورتاق بەھرىمەن بولىدۇ، كېڭىشىپ بىرلىككە كېلىپ يۈرگۈزىدۇ، كېڭىشىپ بىرلىككە كېلەلمىسە ھەم نورمال سەۋەب بولمىسا، ھەرقانداق تەرەپنىڭ يەنە بىر تەرەپنىڭ ئۆتۈنۈپ بېرىش ھوقۇقىدىن باشقا ھوقۇقىنى يۈرگۈزۈشىگە توسقۇنلۇق قىلىشىغا يول قويۇلمايدۇ، لېكىن ئېرىشكەن نەپنى بىرلىشىپ ئاچقۇچىلارنىڭ ھەممىسىگە مۇۋاپىق تەقسىملەپ بېرىش كېرەك. 11 -ماددا باشقىلارنىڭ ھاۋالىسى بىلەن ئېچىپ بېرىلگەن يۇمشاق ماتېرىيال ئەسەر ھوقۇقىنىڭ تەۋەلىكىى ھاۋالە قىلغۇچى بىلەن ھاۋالىنى قوبۇل قىلغۇچى يازما توختامدا پۈتۈشىدۇ، يازما توختام بولمىسا ياكى توختامدا ئېنىق بەلگىلەنمىگەن بولسا، يۇمشاق ماتېرىيال ئەسەر ھوقۇقىدىن ھاۋالىنى قوبۇل قىلغۇچى بەھرىمەن بولىدۇ.

12 -ماددا دۆلەت ئورگانلىرىنىڭ ۋەزىپە چۈشۈرۈشى بىلەن ئېچىلغان يۇمشاق ماتېرىيال ئەسەر ھوقۇقىنىڭ تەۋەلىكى ۋە يۈرگۈزۈلۈشىتۈر ۋەزىپىنامىسىدە ياكى توختامدا بەلگىلىنىدۇ، تۈر ۋەزىپىنامىسىدە ياكى توختامدا ئېنىق بەلگىلەنمىگەن بولسا، يۇمشاق ماتېرىيال ئەسەر ھوقۇقىدىن ۋەزىپىنى قوبۇل قىلغان قانۇنىي ئىگە ياكى باشقا تەشكىلات بەھرىمەن بولىدۇ. 13 -ماددا تەبىئىي شەخس قانۇنىي ئىگە ياكى باشقا تەشكىلاتتا ۋەزىپە ئۆتىگەن مەزگىلدە ئاچقان يۇمشاق ماتېرىيالدا تۆۋەندىكى ئەھۋاللارنىڭ بىرى كۆرۈلسە، بۇ يۇمشاق ماتېرىيالنىڭ ئەسەر ھوقۇقىدىن شۇ قانۇنىي ئىگە ياكى باشقا تەشكىلات بەھرىمەن بولىدۇ، قانۇنىي ئىگە ياكى باشقا تەشكىلات يۇمشاق ماتېرىيالنى ئاچقان تەبىئىي شەخسنى مۇكاپاتلىسا بولىدۇ:

(1)ئۆز خىزمىتىدە ئېنىق بېكىتىلگەن ئېچىش نىشانى بويىچە ئېچىلغان يۇمشاق ماتېرىيال بولسا؛ (2)ئېچىلغان يۇمشاق ماتېرىيال ئۆز خىزمەت پائالىيىتىدە ئالدىن كۆرۈلگەن نەتىجە ياكى تەبىئىي نەتىجە بولسا؛ (3) قانۇنىي ئىگە ياكى باشقا تەشكىلاتلارنىڭ مەبلىغى، مەخسۇس ئۈسكۈنىسى، ئاشكارىلانمىغان مەخسۇس ئۇچۇر قاتارلىق ماددىي ۋە تېخنىكا شارائىتىدىن ئاساسلىق پايدىلىنىپ ئېچىلغان ھەمدە قانۇنىي ئىگە ياكى باشقا تەشكىلات مەسئۇلىيەتنى ئۈستىگە ئالغان يۇمشاق ماتېرىيال بولسا.

14 -ماددا يۇمشاق ماتېرىيال ئەسەر ھوقۇقى يۇمشاق ماتېرىيال ئېچىپ بولۇنغان كۈندىن باشلاپ چۇجۇتقا كېلىدۇ.

تەبىئىي شەخسنىڭ يۇمشاق ماتېرىيال ئەسەر ھوقۇقىنىڭ قوغدىلىش مۇددىتى تەبىئىي شەخسنىڭ ئۆمرىنىڭ ئاخىرىغىچە بولىدۇ ۋە ئۇ ئۆلۈپ 50-يىل 12-ئاينىڭ 31-كۈنى ئاخىرلىشىدۇ، يۇمشاق ماتېرىيال بىرلىشىپ ئېچىلغان بولسا، ئاخىرىدا ئۆلگەن تەبىئىي شەخس ئۆلۈپ 50-يىل 12-ئاينىڭ 31-كۈنى ئاخىرلىشىدۇ.

قانۇنىي ئىگە ياكى باشقا تەشكىلاتنىڭ يۇمشاق ماتېرىيال ئەسەر ھوقۇقىنىڭ قوغدىلىش ھوقۇقى 50 يىل بولىدۇ، يۇمشاق ماتېرىيال تۇنجى قېتىم ئېلان قىلىنىپ 50-يىل 12-ئاينىڭ 31-كۈنى ئاخىرلىشىدۇ، لېكىن يۇمشاق ماتېرىيال ئېچىپ بولۇنغان كۈندىن باشلاپ 50 يىلغىچە ئېلان قىلىنمىسا بۇ نىزام تەرىپىدىن قوغدالمايدۇ. 15 -ماددا يۇمشاق ماتېرىيال ئەسەر ھوقۇقى تەبىئىي شەخسكە تەۋە بولسا، شۇ تەبىئىي شەخس ئۆلگەندىن كېيىن، يۇمشاق ماتېرىيال ئەسەر ھوقۇقىغا ۋارىسلىق قىلغۇچى يۇمشاق ماتېرىيال ئەسەر ھوقۇقىنىڭ قوغدىلىش مۇددىتى ئىچىدە «جۇڭخۇا خەلق جۇمھۇرىيىتىنىڭ ۋارىسلىق قانۇنى»دىكى ئالاقىدار بەلگىلىمە بويىچە بۇ نىزامنىڭ 8-ماددىسىدا بەلگىلەنگەن ئىسمىنى يېزىش ھوقۇقىدىن باشقا ھوقۇقلاردىن بەھرىمەن بولسا بولىدۇ. يۇمشاق ماتېرىيال ئەسەر ھوقۇقى قانۇنىي ئىگە ياكى باشقا تەشكىلاتقا تەۋە بولسا، قانۇنىي ئىگە ۋە باشقا تەشكىلات ئۆزگەرگەن، ئاخىرلاشقاندىن كېيىن، ئەسەر ھوقۇقىدىن مۇشۇ نىزامدا بەلگىلەنگەن قوغداش مۇددىتى ئىچىدە ھوقۇق، مەجبۇرىيىتىنى قوبۇل قىلغان قانۇنىي ئىگە ياكى باشقا تەشكىلات بەھرىمەن بولىدۇ، ھوقۇق-مەجبۇرىيىتىنى قوبۇل قىلغان قانۇنىي ئىگە ياكى باشقا تەشكىلات بولمىسا، دۆلەت بەھرىمەن بولىدۇ. 16 -ماددا يۇمشاق ماتېرىيال قانۇنلۇق كۆپەيتمىسىگە ئىگىدارچىلىق قىلغۇچى تۆۋەندىكى ھوقۇقلاردىن بەھرىمەن بولىدۇ:

(1) ئىشلىتىش ئېھتىياجىغا ئاساسەن شۇ يۇمشاق ماتېرىيالنى كومپيۇتېر قاتارلىق ئۇچۇر بىر تەرەپ قىلىش ئىقتىدارىغا ئىگە قۇرۇلمىغا قاچىلاش؛ (2) كۆپەيتمىنىڭ بۇزۇلۇشىدىن ساقلىنىش ئۈچۈن زاپاس كۆپەيتمە ئىشلەش. بۇ زاپاس كۆپەيتمىنى ھەرقانداق شەكىل بىلەن باشقىلارنىڭ ئىشلىتىشىگە بېرىشكە بولمايدۇ ھەمدە ھەممە ئادەم شۇ قانۇنلۇق كۆپەيتمىگە ئىگىدارچىلىق قىلىش ھوقۇقىدىن مەھرۇم بولغاندا، زاپاس كۆپەيتمىنى كۆيدۈرۈۋېتىشكە مەسئۇل بولىدۇ؛ (3) شۇ يۇمشاق ماتېرىيالنى ئەمەلىي كومپيۇتېر ئىشلىتىش مۇھىتىدا قوللىنىش ياكى ئۇنىڭ ئىقتىدارىنى ياخشىلاش ئۈچۈن زۆرۈر تۈزىتىش ئېلىپ بېرىلىدۇ، لېكىن توختامدا باشقا پۈتۈم بولمىسا، شۇ يۇمشاق ماتېرىيال ئەسەر ھوقۇقى ئىگىسىنىڭ ئىجازىرىتىنى ئالماي تۇرۇپ، ھەرقانداق بىر 3-تەرەپنى تۈزىتىلگەندىن كېيىنكى يۇمشاق ماتېرىيال بىلەن تەمىن ئېتىشكە يول قويۇلمايدۇ. 17 -ماددا يۇمشاق ماتېرىيالدىكى لايىھىلەش ئىدىيىسى ۋە قائىدىسىنى ئۆگىنىش، تەتقىق قىلىش مەقسىتىدە يۇمشاق ماتېرىيالنى قۇراشتۇرۇش، نامايەن قىلىش، يەتكۈزۈش ياكى ساقلاش قاتارلىق شەكىللەر ئارقىلىق ئىشلىتىشتە يۇمشاق ماتېرىيال ئەسەر ھوقۇقى ئىگىسىنىڭ ئىجازىتىنى ئالمىسىمۇ، ھەق تۆلىمىسىمۇ بولىدۇ. ئۈچىنچى باب يۇمشاق ماتېرىيال ئەسەر ھوقۇقىنى ئىجازەتلىك ئىشلىتىش ۋە ئۆتۈنۈپ بېرىش 18 -ماددا باشقىلارنىڭ يۇمشاق ماتېرىيال ئەسەر ھوقۇقىنى ئىشلىتىشىگە ئىجازەت بېرىشتە، ئىجازەتلىك ئىشلىتىش توختامى تۈزۈش كېرەك. ئىجازەتلىك ئىشلىتىش توختامىغا ماتېرىيال ئەسەر ھوقۇقى ئىگىسىنىڭ ئىجازەت بەرگەن ھوقۇقى، ئېنىق يېزىلمىغان بولسا، ئىجازەت بېرىلگۈچىنىڭ ئىشلىتىشىگە بولمايدۇ. 19-ماددا باشقىلارغا يۇمشاق ماتېرىيال ئەسەر ھوقۇقىنى مەخسۇس يۈرگۈزۈش ئىجازىتى بەرگەندە، جاۋابكارلار يازما توختام تۈزۈشى كېرەك.

20 -ماددا يۇمشاق ماتېرىيال ئەسەر ھوقۇقىنى ئۆتۈنۈپ بەرگەندە، جاۋابكارلار يازما توختام تۈزۈشى كېرەك. 21 -ماددا باشقىلارنىڭ يۇمشاق ماتېرىيال ئەسەر ھوقۇقىنى مەخسۇس ئىشلىتىشىگە ئىجازەت بېرىش توختامى ياكى يۇمشاق ماتېرىيال ئەسەر ھوقۇقىنى ئۆتۈنۈپ بېرىش توختامى تۈزگەندە، گوۋۇيۈەننىڭ ئەسەر ھوقۇقىنى مەمۇرىي باشقۇرۇش تارمىقى ئېتىراپ قىلغانيۇمشاق ماتېرىيال رويخەتكە ئېلىش ئورگىنىغا تىزىمغا ئالدۇرسا بولىدۇ. 22 -ماددا جۇڭگو پۇقراسى، قانۇنىي ئىگە ياكى باشقا تەشكىلات چەت ئەللىكلەرگە يۇمشاق ماتېرىيال ئەسەر ھوقۇقىنى ئىشلىتىش ئىجازىتىنى بەرگەندە ياكى ئۆتۈنۈپ بەرگەندە، «جۇڭخۇا خەلق جۇمھۇرىيىتىنىڭ تېخنىكا ئىمپورت-ئېكېسپورتىنى باشقۇرۇش نىزامى»دىكى ئالاقىدار بەلگىلىمىلەرگە رىئايە قىلىشى لازىم.

تۆتىنچى باب قانۇن جاۋابكارلىقى 23 -ماددا »جۇڭخۇا خەلق جۇمھۇرىيىتىنىڭ ئەسەر ھوقۇقى قانۇنىدا«ياكى مۇشۇ نىزامدا باشقا بەلگىلىمە بولماي، تۆۋەندىكى ھوقۇققا دەخلى-تەرۇز قىلىش قىلمىشىنى سادىرقىلغانلار ئەھۋالغا قاراپ دەخلى-تەرۇز قىلىشنى توختىتىش، تەسىرىنى يوقىتىش، ئەپۇ سوراش، زىياننى تۆلەپ بېرىش قاتارلىق ھەق تەلەپ جاۋابكارلىقىنى ئۈستىگە ئېلىشى كېرەك: (1) يۇمشاق ماتېرىيال ئەسەر ھوقۇقى ئىگىسىنىڭ ئىجازىتىنى ئالماي، ئۇنىڭ يۇمشاق ماتېرىيالىنى ئېلان قىلسا ياكى رويخەتكە ئالدۇرسا؛ (2) باشقىلارنىڭ يۇمشاق ماتېرىيالىنى ئۆزىنىڭ قىلىپ ئېلان قىلسا ياكى رويخەتكە ئالدۇرسا؛ (3) ھەمكارلاشقۇچىنىڭ ئىجازىتىنى ئالماي، باشقىلار بىلەن بىرلىشىپ ئاچقان يۇمشاق ماتېرىيالنى ئۆزى مۇستەقىل ئىشلەپ پۈتتۈرگەن يۇمشاق ماتېرىيال قىلىپ ئېلان قىلسا ياكى رويخەتكە ئالدۇرسا؛ (4) باشقىلارنىڭ يۇمشاق ماتېرىيالىغا ئىسمىنى يېزىۋالسا ياكى باشقىلارنىڭ يۇمشاق ماتېرىيالىدىكى ئىسمىنى ئۆزگەرتسە؛ (5)يۇمشاق ماتېرىيال ئەسەر ھوقۇقى ئىگىسىنىڭ ئىجازىتىسىز ئۇنىڭ يۇمشاق ماتېرىيالىنى تۈزەتسە، تەرجىمە قىلسا؛ (6) يۇمشاق ماتېرىيال ئەسەر ھوقۇقىغا دەخلى-تەرۇز قىلغان باشقا قىلمىشلار.

24 -ماددا» جۇڭخۇا خەلق جۇمھۇرىيتىىنىڭ ئەسەر ھوقۇقى قانۇنى،«بۇ نىزام ياكى باشقا قانۇن، مەمۇرىي نىزامدا باشقا بەلگىلىمە بولمىسا، يۇمشاق ماتېرىيال ئەسەر ھوقۇقى ئىگىسىنىڭ ئىجازاىتىنى ئالماي تۆۋەندىكى ھوقۇققا دەخلى-تەرۇز قىلىش قىلمىشلىرىنى سادىر قىلغانلار دەخلى-تەرۇز قىلىشنى توختىتىش، تەسىرىنى يوقىتىش، ئەپۇ سوراش، زىياننى تۆلەپ بېرىش قاتارلىق ھەق تەلەپ جاۋابكارلىقىنى ئۈستىگە ئېلىشى كېرەك، شۇنىڭ بىلەن بىر ۋاقىتتا، جامائەت مەنپەئىتىگە زىيان يەتكۈزگەن بولسا، ئەسەر ھوقۇقىنى مەمۇرىي باشقۇرۇش تارمىقى ھوقۇققا دەخلى-تەرۇز قىلىش قىلمىشىنى توختىتىشقا بۇيرۇيدۇ، قانۇنغا خىلاپ تاپاۋىتىنى مۇسادىرە قىلىدۇ، ھوقۇققا دەخلى-تەرۇز قىلىپ ئىشلىگەن كۆپەيتمىنى مۇسادىرە قىلىدۇ، كۆيدۈرۈۋېتىدۇ، جەرىمانە قويسىمۇ بولىدۇ. ئەھۋال ئېغىر بولسا، ئەسەر ھوقۇقىنى مەمۇرىي باشقۇرۇش تارمىقى ھوقۇققا دەخلى-تەرۇز قىلىپ كۆپەيتمە ئىشلىگەن ئاساسلىق ماتېرىيال، سايمان، ئۈسكۈنىلەرنى مۇسادىرە قىلسا بولىدۇ. جىنايى ئىشلار قانۇنىغا چۈشسە، جىنايى ئىشلار قانۇنىدىكى ئەسەر ھوقۇقىغا دەخلى-تەرۇز قىلىش جىنايىتى، ھوقۇققا دەخلى-تەرۇز قىلىپ ئىشلەنگەن كۆپەيتمىنى سېتىش جىنايىتى توغرىسىدىكى بەلگىلىمىگە ئاساسەن، قانۇن بويىچەجىنايى جاۋابكارلىققا تارتىلىدۇ:

(1)ئەسەر ھوقۇقى ئىگىسىىڭ يۇمشاق ماتېرىيالىنى كۆپەيتكەن ياكى قىسمەن كۆپەيتكەن بولسا؛ (2) ئەسەر ھوقۇقى ئىگىسىىڭ يۇمشاق ماتېرىيالىنى جامائەتكە تارقاتقان، ئىجارىگە بەرگەن، ئۇچۇر تورى ئارقىلىق تارقاتقان بولسا؛ (3) ئەسەر ھوقۇقى ئىگىسىنىڭ يۇمشاق ماتېرىيال ئەسەر ھوقۇقىنى قوغداش ئۈچۈن قوللانغان تېخنىكىلىق تەدبىرىدىن قەستەن ئۆزىنى قاچۇرسا ياكى ئۇنىڭغا بۇزغۇنچىلىق قىلسا؛ (4) يۇمشاق ماتېرىيال ھوقۇقىنى قەستەن ئۆچۈرۈۋېتىپ ياكى ئۆزگەرتىپ، ئېلېكترون ئۇچۇرلىرىنى باشقۇرسا؛ (5)ئەسەر ھوقۇقى ئىگخسىنىڭ يۇمشاق ماتېرىيال ئەسەر ھوقۇقىنى باشقىلارنىڭ يۈرگۈزۈشىگە ئۆتۈنۈپ بەرسە ياكى ئىجازەت بەرسە.

قانۇنى»نىڭ 49-ماددىسىدىكى بەلگىلىمە بويىچە دەۋادىن ئىلگىرى خەلق سوت مەھكىمىسىگە ئالاقىدار قىلمىشنى توختىتىشقا بۇيرۇش ۋە مال-مۈلۈكنى ئامان ساقلاش تەدبىرلىرىنى قوللىنىش توغرىسىدا ئىلتىماس سۇنسا بولىدۇ. 27 -ماددا ھوقۇققا دەخلى-تەرۇز قىلىش قىلمىشىنى توسۇش ئۈچۈن، يۇمشاق ماتېرىيال ئەسەر ھوقۇقى ئىگىسى دەلىل-ئىسپات يوقىلىپ كېتىش ياكى كېيىن ئېرىشىش قىيىن بولۇش ئېھتىمالى بولغان ئھەۋالدا، «جۇڭخۇا خەلق جۇمھۇرىيىتىنىڭ ئەسەر ھوقۇقى قانۇنى»نىڭ 50-ماددىسىدىكى بەلگىلىمە بويىچە، دەۋالىشىشتىن ئىلگىرى خەلق سوت مەھكىمىسىگە دەلىل-ئىسپاتلارنى ئامان ساقلاشنى ئىلتىماس قىلسا بولىدۇ. 28 -ماددا يۇمشاق ماتېرىيال تاللاپ ئىشلىتىشكە بولىدىغان ئىپادىلەش ئۇسۇلى چەكلىك بولۇپ، بار يۇمشاق ماتېرىيال بىلەن ئوخشىشىپ قالسا، بار يۇمشاق ماتېرىيال ئەسەر ھوقۇقىغا دەخلى-تەرۇز قىلغانلىق بولمايدۇ. 30 -ماددا يۇمشاق ماتېرىيال كۆپەيتمىسىگە ئىگە كىشىلەر شۇ يۇمشاق ماتېرىيالنىڭ ھوقۇقىغا دەخلى-تەرۇز قىلىپ ئىشلەنگەن كۆپەيتمە ئىكەنلىكىنى بىلمىسە ياكى بىلىشكە تېگىشلىك مۇۋاپىق سەۋەب بولمىسا، تۆلەم جاۋابكارلىقىنى ئۈستىگە ئالمايدۇ، لېكىن ھوقۇققا دەخلى-تەرۇز قىلىپ ئىشلەنگەن شۇ كۆپەيتمىنى ئىشلىتىشتىن توختىتىشى، كۆيدۈرۈۋېتىشى لازىم. ئەگەر ھوقۇققا دەخلى-تەرۇز قىلىپ ئىشلەنگەن شۇ كۆپەيتمىنى ئىشلەتكۈچى ئېغىر زىيانغا ئۇچرايدىغان بولسا، كۆپەيتمىنى ئىشلەتكۈچى يۇمشاق ماتېرىيال ئەسەر ھوقۇقى ئىگىسى مۇۋاپىق ھەق بەرگەندىن كېيىن داۋاملىق ئىشلەتسە بولىدۇ. 31 -ماددا يۇمشاق ماتېرىيال ئەسەر ھوقۇقىغا دەخلى-تەرۇز قىلىش ماجىرالىرىنى ياراشتۇرۇشقا بولىدۇ.

يۇمشاق ماتېرىيال ئەسەر ھوقۇقى توختام ماجىراسىنى توختامدىكى كېسىم ماددىلىرى ياكى كېيىن ھاسىل قىلغان يازما كېسىم كېلىشىمىگە ئاساسەن كېسىم ئاپپاراتىغا كېسىم قىلىپ بېرىشنى ئىلتىماس قىلسا بولىدۇ.

ئالاقىدارلارنىڭ توختامدا كېسىم ماددىلىرى بولمىسا، كېيىن يازما كېسىم كېلىشىمىمۇ ھاسىل قىلمىغان بولسا، خەلق سوت مەھكىمىسىگە بىۋاسىتە دەۋا قىلسا بولىدۇ.

بەشىنچى باب قوشۇمچە پرىنسىپ 32 -ماددا بۇ نىزام يولغا قويۇلۇشتىن ئىلگىرى يۈز بەرگەن دەخلى-تەرۇز قىلىش قىلمىشى شۇ قىلمىش سادىر بولغان چاغدىكى دۆلەتنىڭ ھوقۇققا دەخلى-تەرۇز قىلىش قىلمىشىغا ئالاقىدار بەلگىلىمىسى بويىچە بىر تەرەپ قىلىنىدۇ.

33 -ماددا بۇ نىزام 2002-يىل 1-ئاينىڭ 1-كۈنىدىن باشلاپ يولغا قويۇلىدۇ. 1991-يىل 6-ئاينىڭ 4-كۈنى گوۋۇيۈەن ئېلان قىلغان كومپيۇتېر يۇمشاق ماتېرىيالىنى قوغداش نىزامى ئەمەلدىن قالدۇرۇلىدۇ
