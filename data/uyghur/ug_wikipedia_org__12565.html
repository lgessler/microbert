مۇسلىمە ئاياللارنىڭ ئىسلامدىكى ئورنى مۇسلىمە ئاياللارنىڭ ئىسلامدىكى ئورنى

رەسۇلۇللاھ سەللەللاھۇ ئەلەيھى ۋەسەللەم مۇنداق دەيدۇ: «سالىھە ئايال دۇنيادىكى بارلىق نېمەتلەرنىڭ ئەڭ ئېسىلىدۇر» [مۇسلىم رىۋايىتى].

ئىسلام دىنى ئاياللار مەسىلىسىگە ئالاھىدە ئەھمىيەت بېرىدىغان دىن بولۇپ، تارىختا ئۆتكەن مەشھۇر مۇسلىمە ئاياللار بىلەن ئۇيغۇر قىز-خانىملىرىنىڭ ھەقىقەتنى ياقىلاش يولىدا كۆرسەتكەن تۆھپىلىرىدىكى ئوخشاشلىق، ئۇيغۇر مۇسلىمە ئاياللارنىڭ دىن ۋە مىللەتنى قوغداش يولىدا ھەرۋاقىت بەدەل تولەشكە تەييار ئىكەنلىكىنى تەكىتلەيدۇ . 1- ئاياللار ئۇلۇغ:

بارچە كائىناتنىڭ ياراتقۇچىسى بولغان قۇدرەتلىك ئاللاھ، ئىنسانلارنىڭ ئاتىسى ئادەم ئەلەيھىسسالامنى ياراتقاندىن كېيىن، ئۇنىڭ جۈپتى ھاۋۋانى ئادەم ئاتىنىڭ چەپ قوۋۇرغىسىدىن ياراتتى ۋە ئۇنى ئادەم ئاتىغا ئۆمۈرلۈك ھەمراھ بولۇشقا بۇيرىدى.

ئاللاھنىڭ بۇخىل ئورۇنلاشتۇرۇش قانۇنىيىتى نەتىجىسىدە، ئاتىمىز ئادەم بىلەن ئانىمىز ھەۋۋا بىر-بىرلىرىگە جەننەت ۋە زېمىندا ئۆمۈرلۈك ھەمراھ بۇلۇپ، نەتىجىدە زېمىننىڭ مىراسخورلىرى بولىدىغان بىز ئىنسانلارنىڭ كۆپىيىشىدىكى ئاساسلىق مەنبە، شۇنداقلا ئىنسانىيەتنىڭ ئانىسىغا ئايلىنىش شەرىپىگە نائىل بولۇشى ئاياللارنىڭ نەقەدەر ئۇلۇغ ئىكەنلىكىنى بىلدۇرىدۇ.

ئاياللارنىڭ ئۇلۇغلىقى شۇكى؛ بارچە مۇسۇلمانلار ئۇلۇغلايدىغان ۋە قىممەتلىك ئابى-ھايات دەپ بىلىدىغان بەرىكەتلىك زەمزەم قۇدۇقىنىڭ، بىر ئايال سەۋەبلىك بارلىققا كېلىشى ۋە قىيامەتكىچە ئىنسانلارنىڭ ئەبەدىي قەدىرلەيدىغان مۆجىزىلىك بۇلىقىغا ئايلىنىشى. ئاياللارنىڭ ئۇلۇغلىقى شۇكى؛ ئاللاھنىڭ ئەڭ بۈيۈك مۆجىزىلىرىدىن بىرى ئاتىسىز يارىتىلغان پەيغەمبەرنىڭ ئانىسى(بۇ يەردە مەريەم ئانىمىز كۆزدە تۇتىلىدۇ-م) بولۇشتەك بۈيۈك مەرتىۋىگە سازاۋەر بولىشى ۋە بۇ مۆجىزىدىن كېيىنكى ئېغىر تۆھمەت-ھاقارەتلەرگە چىرايلىقچە سەبرى قىلىپ، ئېغىر سىناقلاردىن مۇۋەپپەقىيەتلىك ئۆتۈش ئارقىلىق قىيامەتكە قەدەر ئىنسانلار تەرىپىدىن مەدھىيەلىنىشى ۋە ئۇلارنىڭ ھۆرمەتلىك ئانىسىغا ئايلىنىشى. ئاياللارنىڭ ئۇلۇغلىقى شۇكى؛ ئاللاھنىڭ ئىرادىسى بىلەن، سۆيۈملۈك پەيغەمبىرىمىز مۇھەممەد ئەلەيھىسسالامغا پەيغەمبەرلىك ۋەھىيىسى باشلانغان بىرىنچى كۈنىدە، ئۇنىڭ پەيغەمبەرلىكىنى ئېتىراپ قىلغان ۋە ئۇنىڭغا ئەۋۋەل ئىمان كەلتۈرگەن شەخىس، رەسۇلۇللاھنىڭ ئايالى ۋە بارچە مۇسۇلمانلارنىڭ ھۆرمەتلىك ئانىسى خەدىچە رەزىيەللاھۇ ئەنھا بولۇشتەك ئىسلام تارىخىدا بىرىنچىلىك ئورۇننى ئىگەللىشى. ئاياللارنىڭ ئۇلۇغلىقى شۇكى؛ سۆيۈملۈك پەيغەمبىرىمىز ئېلىپ كەلگەن مۇقەددەس ئىسلام دىننىڭ بارچە ئەھكاملىرىنى، ئۇنىڭ ۋاپاتىدىن كېيىن بىز مۇسۇلمانلارغا يەتكۈزۈشتە، ئاساسلىق ۋاستىچى بولغان كىشى يەنىلا بارچە مۆمىنلەرنىڭ ئانىسى بولۇشتەك شەرەپلىك مەرتىۋىگە مۇيەسسەر بولغان ئائىشە رەزىيەللاھۇ ئەنھادەك پىقھىشۇناس ئالىمەنىڭ ئاياللاردىن بولىشى. ئاياللارنىڭ ئۇلۇغلىقى شۇكى؛ مۇھەممەد ئەلەيھىسسالام پەيغەمبەرلىك ۋەزىپىسىنى باشلىغان دەۋرىدە، ئىسلام دۈشمەنلىرىنىڭ رەسۇلۇللاھقا قىلغان ئېغىر ئەزىيەتلىرى، (يەنى ئەقبە ئىبنى مۇئېيت باشچىلىقىدىكى بىر ئۈچۈم مەككە مۇشرىكلىرى رەسۇلۇللاھ سەجدىدىكى ۋاقىتتا ئۈستىگە تۆگىنىڭ ئۈچەي-قېرىنلىرىنى تاشلاشتەك) شۇ ئېغىر ئاھانەتلىك ۋاقىتلاردا كىچىكلىكىگە قارىماي كۇففارلار ئالدىدا جاسارەت قىلىپ، رەسۇلۇللاھقا ياردەم بېرىشتە يەنىلا بىرىنچى بۇلۇپ مەيدانغا سەكرەپ چىققىنى ئۇنىڭ سۆيۈملۈك قىزى پاتىمە رەزىيەللاھۇ ئەنھادۇر.

رەسۇلۇللاھ سەللەللاھۇ ئەلەيھى ۋەسەللەم بۈيۈك ئىسلام ھاكىمىيىتىنى قۇرۇپ، بارچە مۇسۇلمانلارغا يېتەكچىلىك قىلىۋاتقان دەۋرىگە كەلگەندىمۇ، كۈندىلىك تۇرمۇشىنىڭ غۇربەت ئىچىدە ئۆتىشىگە قارىماستىن، ئۆزلىرى ئاچ قالسىمۇ قولىدىكىنى باشقىلارغا ئۆتۈنۈپ بېرىشتەك ئالىيجانابلىق بىلەن سۈپەتلەنگۈچى ۋە بۇ دۇنيا بىلەن خوشلىشىشتىن بۇرۇنلا جەننەت بىلەن بىشارەت بىرىلگەن شەخسمۇ دەل مۇسلىمىلەرنىڭ ئۈلگىلىك ئانىسى پاتىمە رەزىيەللاھۇ ئەنھادۇر.

ئاياللارنىڭ ئۇلۇغلىقى شۇكى؛ سۆيۈملۈك پەيغەمبىرىمىز مۇھەممەد ئەلەيھىسسالام ئېلىپ كەلگەن بۇ مۇقەددەس ئىسلام دىنىنى قوغداش يولىدا، تۈرلۈك زۇلۇم-ئىسكەنجىلەرگە قارشى ئۆز ھاياتىدىن ئايرىلىشقا تەييار تۇرۇپ، ئىسلام ئېتىقادىدىن ئەسلا ۋاز كەچمەسلىكتەك ھەقىقەتتە مۇستەھكەم تۇرۇپ، ئىمانىنى ساقلاش يولىدا جان بەرگەن ئىسلامدىكى تۈنجى شەھىدە ۋەئىسلام ئاياللىرىغا ئىماندا قانداق مۇستەھكەم تۇرۇش كېرەكلىكىنى ئەمەلىيىتى بىلەن كۆرسىتىپ بەرگەن ئانىمىز سۇمەييە رەزىيەللاھۇ ئەنھادۇر. ئاياللارنىڭ ئۇلۇغلىقى شۇكى؛ مۇقەددەس ئىسلام دىنىنى قوغداش ۋە دىن دۈشمەنلىرىنىڭ تاجاۋۇزىغا قارشى ئېلىپ بېرىلغان جەڭلەردە، بىر تۈركۈم ئاياللارنىڭمۇ جىھادقا بىرلىكتە چىقىشى ۋە ئۇلارنىڭ جەڭ مەيدانلىرىدا رەسۇلۇللاھنى ئۆز جېنىدىنمۇ ئەزىز ۋە چارە كۆرۈپ ھىمايە قىلىشلىرى، شۇنداقلا ئاللاھنىڭ رازىلىقىغا ئېرىشىشنى مەقسەت قىلغان ئونلىغان-يۈزلىگەن ئانىلار، ئۆزلىرىنىڭ ياش-نەۋقىران ئوغۇللىرىنى رەسۇلۇللاھنىڭ ئېتىرازىغا قارىماي جىھادقا يوللىشى ۋە ئوغۇللىرىنىڭ شېھىت بولغانلىق خەۋىرىنى ئاڭلىغان ھامان ئاللاھقا شۈكرانە سەجدىلەرنى قىلىپ، مەن ئوغلۇمنى ئەنە شۇنىڭ ئۈچۈن بېقىپ يېتىشتۈرگەنمەن، مەن ئوغلۇمغا بەرگەن ئاق سۇتۇمگە مانا ئەمدى رازى بولدۇم دېيىشلىرى ئارقىلىق، پەيغەمبىرىمىزنىڭ شاپائەتلىك دۇئالىرىغا مۇشەررەپ بولۇشلىرى. ئاياللارنىڭ ئۇلۇغلىقى شۇكى؛ سۆيۈملۈك پەيغەمبىرىمىز مۇھەممەد ئەلەيھىسسالامغا تۈنجى ئىمان كەلتۈرگەنلەرنىڭ بىرىنچىسىمۇ ئايال، ئىسلام دىنىنى قوغداش يولىدا ئەۋۋەل جان بەرگەنلەرنىڭ بىرىنچىسمۇ ئايال بولىشى نەتىجىسىدە، مۇقەددەس كىتابىمىز قۇرئان كەرىمدىكى بىر سۈرىنىڭ ئىسمى «سۈرە نىسا» (ئاياللار سۈرىسى) دېگەن نام بىلەن ئاتىلىپ، ئاياللار بىلەن ئاللاھ ئوتتۇرسىدىكى بۈيۈك ئىلاھى باغلانتىنىڭ قىيامەتكە قەدەر باقى قېلىشى.

ئەنە شۇ بۈيۈك مەرتىۋىلەرنىڭ جەۋھىرى بولغان بىشارەت-پەيغەمبەر ئەلەيھىسسالامنىڭ مۇبارەك ۋەسىيەتلىرىدىكى «..جەننەت ئاناڭنىڭ ئايىغى ئاستىدا..» [نەسائىي رىۋايىتى 3104-ھەدىسنىڭ بىر قىسمى] دېگەن ئۇلۇغ شەرەپكە نائىل بۇلىشى.

يۇقىرىقى شەرەپ ۋە مەرتىۋىلەرنىڭ پەقەت ئاياللارغىلا نېسىپ بولغانلىقى، ئىسلام ئاياللىرىنىڭ نەقەدەر ئىززەت-ھۆرمەتكە سازاۋەر ئىكەنلىكىنى كۆرسەتمەكتە.

2 - ئاياللار دىننىڭ يېرىمى:

سۆيۈملۈك پەيغەمبىرىمىز مۇھەممەد、ئەلەيھىسسالام: «كىمكى بىر سالىھە ئايال بىلەن ئۆيلەنسە ئۇ كىشى دىننىڭ يېرىمىنى تولدۇرغان بولىدۇ. قالغان يېرىمىدا ئاللاھتىن قورقسۇن» دېگەن. [ھاكىم ۋە زەھەبى رىۋايىتى].

ئاياللار دىننىڭ يېرىمى دېگىنىمىز شۇكى؛ ئايالى بار ئەر بىلەن، بويتاق ئەرنىڭ كۈندىلىك تۇرمۇش ھاياتى ھەرگىز ئوخشاش بولمايدۇ، ئايالسىز ئەر ھەرقانچە تەقۋا بولسىمۇ، يەنىلا بەزى كۆز سەھۋەنلىكلىرىدىن خالى بولالماسلىقى مۇمكىن. ئەمما تەقۋادار، ئىپپەتلىك بىر ئايالى بار ئەرگە بەزى خاتالىقلاردىن ئۆزىنى تارتىشقا ئايالىنىڭ چوڭ ياردىمى بولىدۇ. ئەنە شۇ تەرەپلەرنى نەزەردە تۇتقاندىمۇ ئاياللار دىننىڭ يېرىمى بۇلۇپ ھېسابلىنىدۇ.

3 -ئاياللار ئائىلىنىڭ باھارى:

پاك دىيانەتلىك ئايال نىكاھلانغاندىن كىيىن ئۇستىگە يۈكلەنگەن ئائىلە مەسئۇلىيىتىنى تونۇيدۇ. يولدىشى ۋە پەرزەنتلىرىنىڭ خوشاللىقى ئۈچۈن ھاياتىنى بېغىشلايدۇ. قانائەتچانلىق بىلەن ئازادە-رەتلىك تۇرمۇش مۇھىتى يارىتىش ئارقىلىق، ئائىلىسىگە بەخت-سائادەت ئاتا قىلىدۇ.

رەسۇلۇللاھ ئەلەيھىسسالامدىن پەزىلەتلىك ئاياللار توغرىسىدا سورالغىنىدا: «ئېرىگە ئىتائەتمەن، قارىسا قەلبىگە راھەت بېغىشلايدىغان (يەنى ئىرى ئۈچۈن پەردازلىنىدىغان ۋە چىرايلىق كىيىنىدىغان)، ئىرى يوق ۋاقىتلاردا يولدىشىنىڭ ھەققى ۋە پۇل-مېلىغا خىيانەت قىلمايدىغان ئامانەتدار ئاياللار» دەپ جاۋاب بەرگەن. [ئەھمەد ۋە ھاكىم رىۋايىتى].

4-ئاياللار ئۆينىڭ بەرىكىتى:

كىمىكى ئىپپەتلىنىشنى ئۈمىد قىلىپ نىكاھلانسا، ئاللاھ ئۇ ئەر-ئايالنىڭ تۇرمۇشىغا بەرىكەت ئاتا قىلىدۇ. بۇ ھەقتە ئاللاھ تائالا مۇنداق دەيدۇ: «ئاللاھ ئۇلارنى ئۆز كەرىمى بىلەن باي قىلىدۇ، ئاللاھنىڭ مەرھىمىتى كەڭدۇر، ئاللاھ بەندىلەرنىڭ مەنپەئىتىنى بىلگۈچىدۇر.» [سۈرە نۇر 32-ئايەتنىڭ بىر قىسمى].

پەرۋەردىگارىنىڭ ھەققىنى تونۇغان ئايال ئەلۋەتتە ئىرى ۋە پەرزەنتلىرىنىڭ ھەققىنى تونۇيدۇ. قانائەتچانلىق بىلەن تۇرمۇش كەچۈرۈشنى ئادەتكە ئايلاندۇرۇپ، يولدىشىنىڭ قۇربىتى يەتمىگەن نەرسىلەرنى تەلەپ قىلماستىن، ئائىلە ئىقتىسادىغا ماس ھالدا تۇرمۇشىنى تەرتىپكە سالىدۇ.

5-ئاياللار يېرىم دۇنيا:

پەرزەنتلىرىنى ئىمانى ئاڭ مىللى روھ، ۋەتەن سۆيگۈسى چۈشەنچىسى بىلەن تەربىيىلەش ئارقىلىق، ئىنساندەك ياشاشنى بىلىدىغان، ھەقىقەتپەرۋەر، زۇلۇمغا سۇكۇت قىلىشنى نومۇس دەپ بىلىدىغان قەھرىمان، جاسارەتلىك ئوغلانلارنى يېتىشتۈرگەن ۋە يېتىشتۇرىۋاتقان تەقۋادار ۋە ئىماندار، غايىلىك ئاياللار ھەرۋاقىت ۋە ھەرزامان جەمئىيەت تەرەققىياتىغا ئۈزلۈكسىز تۆھپىلەرنى قوشىدۇ. ئۇلارغا ئاللاھنىڭ خۇش بېشارىتى بولسۇن.

ئاياللار جۈملىسىدىن بولغان بىز ئۇيغۇر مۇسۇلمانلىرىنىڭ قىز-خانىملىرىمۇ، ئۇلۇغلۇقتا، دىننىڭ يېرىمى بولۇشتا، ئائىلىنىڭ گۈزەللىكى بولۇشتا، ئۆينىڭ بەرىكىتى بولۇشتا، ھەتتا دۇنيانىڭ يېرىمى بولۇشتىمۇ باشقا مۇسۇلمان ئاياللاردىن ھەرگىز قېلىشمايدۇ، بەلكى ئۇلارنىڭ ئالدىنقى قاتارىدا تۇرغانلىقى ۋە تۇرىدىغانلىقىدا شەك يوق. چۈنكى يىراق تارىخقا بارماستىن ئۆزىمىزنىڭ يېقىنقى ئىككى-ئۈچ ئەسىرلىك تارىخىمىزغا قارايدىغان بولساق، ئۇيغۇر ئاياللىرىنىڭ نەقەدەر ئۇلۇغ، نەقەدەر دىندار، نەقەدەر گۈزەل، نەقەدەر بەرىكەتلىك ئىكەنلىكىنى روشەن كۆرۈۋالالايمىز.

بۈيۈك خاندانلىقىمىز ئاغدۇرۇلۇپ ئۇزۇن ئۆتمەي، يۇرتىمىزنىڭ ھەرقايسى جايلىرىدا تاجاۋۇز ۋە زوراۋانلىقلارغا قارشى ئېلىپ بېرىلغان خەلق قوزغىلاڭلىرىدا، ئۇيغۇر ئاياللىرىنىڭ ئالدىنقى سەپتە تۇرۇپ، كۈرەش مەيدانلىرىدا ھاياتىنىڭ ئاخىرغىچە تەۋرەنمەستىن، ھەقىقى ئىسلام ئاياللىرىغا خاس ئەنئەنىۋى جاسارىتى بىلەن، ئۆلسەك شېھىت قالساق غازى دەپ مەردانىلەرچە ئېلىشىپ، قەھرىمانلارچە شېھىت بولغانلىقلىرىدەك تارىخى ئەسلىمىلەرنى 1765-يىلدىكى مەنچىڭ ھۆكۈمرانلىقىغا قارشى ئۇچتۇرپاندا يۈز بەرگەن خەلق ئىنقىلابىنىڭ ئۇنتۇلماس سىماسى بولغان "يەتتە قىزلار مەقبەرىسى" سىمۋولىدىن كۆرۈۋالالايمىز.

مەزكۇر ئىنقىلابتىن كېيىنمۇ ئاللاھنىڭ پەزلى بىلەن مىللىتىمىز ئىچىدىن كۆپلىگەن ئىماندار، جەڭگىۋار، قەھرىمان قىز-خانىملار ئونلاپ-يۈزلەپ بارلىققا كەلدى ۋە بۇنىڭدىن كېيىنمۇ كېلىدۇ. ئۇلار مىللىتىمىزنىڭ دىنى ۋە مىللى كىملىكىنى ساقلاپ قېلىش، دىنى ۋە مىللى مەدەنىيەتنى قوغداش يولىدا ئاغزىدىن جان، تومۇرلىرىدىن قان بەرسە-بەردىكى، مۇقەددەس ئىسلامغا بولغان ئىشەنچ-ئېتىقادىنى قەتئى قولدىن بەرمىدى. بۇنىڭ ئەمىلى مىساللىرىنىمۇ، يېقىنقى يىللاردىن بىرى خوتەندە، قاغىلىقتا، يەكەندە، پەيزىۋاتتا، ئاقسۇدا، ئۈچتۇرپاندا، كۇچاردا، شاياردا ۋە ئېلىدا يۈزبەرگەن، چوڭ-كىچىك ئاياللار ھەرىكىتىدىن روشەن كۆرۈۋالالايمىز.

ئەنە شۇ ئۇلۇغ ئۇيغۇر ئاياللىرى بۈيۈك ئەقىدە ۋە جەڭگىۋارلىقنى لازىم تۇتقانلىقى نەتىجىسىدە، ئەڭ يېقىن كۈنلەردە يۈزبەرگەن 5-ئىيۇل خەلق قوزغىلىڭى جەريانىدىمۇ، يەنىلا خانىم-قىزلىرىمىز سەپنىڭ ئالدىدا ماڭدى . ئىمانى ۋولقان، ۋىجدانى جاسارەتلىرى بىلەن ئۇيغۇرلارنى پۈتۈن دۇنياغا قايتىدىن تونۇتتى. شۇنداقلا ئۆزلىرىنىڭ قەدىمدىنلا دۇنيا مەدەنىيىتىگە بۈيۈك تۆھپىلەرنى قوشقان ئۇلۇغ مۇسۇلمان ئاياللار تەبىقىسىدىن ئىكەنلىكىنى ۋە ۋاقتى كەلگەندە ھەقىقەت ئۈچۈن بارلىقىنى قۇربان قىلىشقا تەييار ئىكەنلىكىنى قايتا نامايەندە قىلدى .مىللىتىمىز مانا مۇشۇنداق ھەقىقى ئىماندار، ھاياتىنى ھەقىقەت ئۈچۈن بېغىشلىيالايدىغان، قورقماس ۋە جەڭگىۋار ئانىلارغا مۇھتاج.

ئەتراپىمىزغا قارايدىغان بولساق، يېقىنقى يىللاردىن بۇيان مۇسۇلمان مىللىتىمىزنىڭ كېلىچىكى ئۈستىدە كۆپرەك ئويلىنىدىغانلارمۇ يەنىلا مىللىتىمىز ئىچىدىن چىققان دىندار، ھەقىقەتپەرۋەر، بەرىكەتلىك ئاياللار تەبىقىسىدىن بولماقتا!. خۇسۇسەن چەتئەلدىكى قىز ئوقۇغۇچىلارنىڭ سانىنىڭ چەكلىكلىكىگە قارىماي، ئۇيغۇر قىزلىرىنىڭ ئىلىمغا ۋە ئىلىم ئوقۇشتىكى ئەسلى غايىسىگە بولغان ساداقەتمەنلىك تۇيغۇسى كۈنسايىن يۈكسەلمەكتە. ھەمدە ئۇلار ئۆزلىرىنىڭ ئۇقۇش مۇددىئاسى ۋە ئۇقۇش پۈتتۈرگەندىن كېيىنكى مەقسەتلىرىنى ھازىرنىڭ ئۆزىدىلا مۇئەييەنلەشتۈرمەكتە!.

ھەر ساھەدە ئىلىم تەھسىل قىلىۋاتقان قىزلارنىڭ، ئوغۇل ئوقۇغۇچىلارغا قارىغاندا دىن، ۋەتەن-مىللەتكە بولغان سۆيگۈ-مۇھەببىتى شۇنداقلا ھەقىقەتكە ئىنتىلىشى ئالاھىدە پەرقلىق بولماقتا. مىسالغا ئالساق: مەلۇم بىر دۆلەتتىكى ئۇيغۇر مۇھاجىر ئاياللىرىنىڭ ئىسلامغا بولغان مۇستەھكەم ئېتىقادى، دىنى تەلىم-تەربىيەگە بولغان قىزىقىشى، ۋەتەن-مىللەتكە بولغان قايغۇرۇش ۋە ئېچىنىشلىرىغا نىسبەتەن، شۇ يەردىكى ئەرلەرنىڭ دىنى بىلىملەرگە بولغان ھېرىسمەنلىكى ۋە ئىجابىي كۆز قاراشلىرى، ۋەتەن-مىللەتنىڭ قايغۇسىدا پىكىر يۈرگۈزۈش قاتارلىق روھى ئويغۇنىشلىرى خىلىلا پەرقلىق ئورۇندا تۇرىدىغانلىقى بۇنىڭ تىپىك مىسالى بولالايدۇ.

ۋەتەن ئىچى ۋە سىرتىدا دىنى ساھەدە سىستىملىق ياكى سىستىمسىز ئىلىم تەھسىل قىلىۋاتقان قىز ئوقۇغۇچىلار، ئۆزلىرىنىڭ دىنى ۋە مىللى مەجبۇرىيەتلىرىنى تۇلۇق تونۇپ، ۋاقتىنى بىكار ئۆتكۈزمەستىن كېچە-كۈندۈز تىرىشىپ، ئىسلامغا ۋە مىللىتىمىزگە پايدىلىق تۈرلۈك مەزمۇنلاردا، ناھايىتى قىممەتلىك ئىلمى ئىجادىي ۋە تەرجىمە ئەسەرلەرنى تەييارلاپ، مېدىيا ۋاستىلىرى ئارقىلىق خەلقىمىزنىڭ نۆۋەتتىكى ئېھتىياجلىرىغا ئازراق بولسىمۇ مەنىۋى ئوزۇق بېىغىشلىماقتا . ئۇلارنىڭ شارائىتلىرى يار بەرگەن دەرىجىدە دىننى يەتكۈزۈشتە كۆرسەتكەن تىرىشچانلىقىنى، بەزى ئەركەك ئوقۇغۇچىلار ياكى دەۋەتچىلەرنىڭ ئۆز مەجبۇرىيەتلىرىنى تونۇشىغا سېلىشتۇرغاندا، ئوغۇل ئوقۇغۇچىلىرىمىز ياكى دەۋەتچىلىرىمىز يەنىلا پاسسىپ ھالدا تۇرۇۋاتقانلىقىنى كېلىۋاتقانلىقىنى كۆرۈۋالالايمىز.

قاچانكى بىز، ئانىلىرىمىزنىڭ بىزنى ساغلام ئەقىدە ۋە جەڭگىۋارلىق روھى بىلەن تەربىيەلەپ يېتىشتۇرىشىدىكى ئالى مەقسەتلىرىنى تۇلۇق بېلىپ يېتىدىكەنمىز، ئىنسانلارنىڭ ئۇلۇغى، دىنىمىزنىڭ يېرىمى، ئائىلىمىزنىڭ چىرىغى، ئۆيىمىزنىڭ بەرىكىتى بولغان ئانىلىرىمىزنىڭ ئۈمىدىنى ھەقىقى تۈردە ئاقلاشنى، ئىمانى ۋە ۋىجدانى مەجبۇرىيەت دەپ تونۇيمىز .

بۇ مەجبۇرىيەتلەرنى تۇلۇق ئادا قىلىشىمىز ئۈچۈن، ئەنە شۇ بۈيۈك پەزىلەتلەرگە ساھىب بولغان ۋە كەلگۈسىدە ساھىپ بۇلۇش ئالدىدا تۇرىۋاتقان قىز-خانىملىرىمىزنىڭ، ئىنسانلاردۇنياسىدىكى مەرتىۋىسى ۋە ئىسلامدىكى يۈكسەك ئورنىنى ھەرگىز ئېسىمىزدىن چىقارماسلىقىمىز كېرەك!.
